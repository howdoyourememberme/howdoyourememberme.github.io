var ul = document.getElementById(“lists”);
var li1 = document.getElementById(“li1”);
var li2 = document.getElementById(“li2”);
var li3 = document.getElementById(“li3”);

// ###事件委托：通俗地说，就是某个事件本来该自己干的，但是自己不干，交给别人来干。就叫事件委托。
// 举个例子：一个button对象，本来自己需要监控自身的点击事件，但是自己不来监控这个点击事件，让自己的父节点来监控自己的点击事件。




// 传统的做法像前面的这个这样，为每个li添加一个点击事件，但是如果点击个数太多，
// 所有的元素都采用这种方式的话，代码就会太多太多，性能也不高，

// 法1
// li1.addEventListener("click",function(){
//   console.log(1);
// },false);
// li2.addEventListener("click",function(){
//   console.log(2);
// },false);
// li3.addEventListener("click",function(){
//   console.log(3);
// },false);

// 所以我们可以采用事件委托解决这个问题，只需要将事件委托给该一个元素即可
// 事件委托是利用事件冒泡原理，将事件添加给ul;
// 内部的所有子元素均会在执行时触发冒泡，从而执行ul的点击事件效果

 ul.addEventListener("click",function(event){
      // 这个法2,如果不看后面的判断句,
      // ul内部的所有子元素均可触发这个点击事件，
      // 但是我们现在只想要第一个li标签触发
      // 那么就得找到这个li
      // 所以呢，这里需要使用事件对象，所以在这个函数里添加一个参数event
      // 大家可能会想到用this，但不能找到这个真正触发事件的元素,找的是ul
      //
      // event.target 可以获取到真正触发事件的元素，在IE低版本,是用event.srcElement
      console.log(event);
      //
      //这里需要考虑兼容性
      var target = event.target || event.srcElement;


      console.log(event.target);

      console.log(this);
      // 大家可以都打印下看看，在看下面，

      // 这里if()else()语句是判断究竟是点击的哪个元素执行的
      // 如果是第一个li,就执行打印了1
      // 第二个的话就打印了2

      if(target == li1){

        console.log(li1);

       console.log("1");

       }
       if(target == li2){

          console.log("2");

       }
       if(target == li3){

          console.log("3");

       }
    })
